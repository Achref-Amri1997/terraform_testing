provider "aws" {
  region = var.aws_region  
}

resource "aws_s3_bucket" "my_bucket" {
  bucket = "my-unique-bucket-name"  # Replace with a unique bucket name
  acl    = "private"

  tags = {
    Name        = "MyBucket"
    Environment = "Test"
  }
}

variable "aws_region" {
  description = "The AWS region to deploy resources"
  type        = string
}

output "bucket_id" {
  value = aws_s3_bucket.my_bucket.id
}
